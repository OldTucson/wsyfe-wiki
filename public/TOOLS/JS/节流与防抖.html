    # 防抖:

    任务频繁触发的情况下，只有任务触发的间隔超过指定间隔的时候，任务才会执行。

    应用: 在接口防重复调用等等情况中应用多

    /**
 * 防抖的简单原理实现
 * @param {*} fn  防抖后要执行的函数
 * @param {*} time  防抖时间(ms)
 */
function debounce(fn, time) {
    let timeout = null;
    return function() {
      clearTimeout(timeout);
      timeout = setTimeout(() => {
        fn.call(this, arguments);
      }, time);
    };
  }

    # 节流:

    指定时间间隔内只会执行一次任务。

    应用: 在懒加载计算监听时等等情况中应用多

    
/**
 * 节流的简单实现,返回是函数调用的结果
 * @param {*} fn 要节流的函数
 * @param {*} time 需要节流的毫秒。
 */
function throttle(fn,time) {
	let canRun = true;
	return function() {
		if (!canRun) {
			return;
		}
		canRun = false;
		setTimeout(() => {
			fn.call(this, arguments);
			canRun = true;
		}, time);
	};
}
